Tests p:document with a simple text document in UTF-8 (no charset!).

Test ab-p-document020.xml is expected to pass.

The pipeline


      <p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
            <p:output port="result"/>        

            <p:identity>
                <p:with-input>
                    <p:document href="../documents/text-file-utf-8.txt" content-type="text/plain"/>
                </p:with-input>
            </p:identity>
            <p:wrap-sequence wrapper="doc"/>
            
        </p:declare-step>
